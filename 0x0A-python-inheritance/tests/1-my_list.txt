The ``1-my_list`` module
======================

>>> MyList = __import__('1-my_list').MyList

>>> my_list = MyList()
>>> type(my_list) == MyList
True

my_list = MyList()
assert issubclass(MyList, list)

my_list.append(5)
assert my_list == [5]

my_list.append(3)
my_list.append(1)
my_list.print_sorted()
[1, 3, 5]

# Test case 4: Check if print_sorted()
my_list.append(-2)
my_list.append(-5)
my_list.print_sorted()
[-5, -2, 1, 3, 5]

empty_list = MyList()
empty_list.print_sorted()
[]

sorted_list = my_list.print_sorted()
assert sorted_list == [-5, -2, 1, 3, 5]
assert sorted_list is not my_list
